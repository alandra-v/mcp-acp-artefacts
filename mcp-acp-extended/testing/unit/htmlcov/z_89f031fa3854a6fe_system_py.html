1"""Pydantic models for system/operational logs. 

2 

3IMPORTANT: The 'time' field in all models is Optional[str] = None because: 

4- Model instances are created WITHOUT timestamps (time=None) 

5- ISO8601Formatter adds the timestamp during log serialization 

6- This provides a single source of truth for timestamps 

7- Logged events ALWAYS have a 'time' field in ISO 8601 format (e.g., "2025-12-11T10:30:45.123Z") 

8 

9This separation means: 

10- Models define the structured event DATA 

11- Formatter adds the TIMESTAMP during serialization 

12- No duplicate timestamp generation across the codebase 

13""" 

14 

15from __future__ import annotations 

16 

17__all__ = [ 

18 "ConfigHistoryEvent", 

19 "PolicyHistoryEvent", 

20 "SystemEvent", 

21] 

22 

23from typing import Any, Dict, Literal, Optional 

24 

25from pydantic import BaseModel, ConfigDict, Field

28class SystemEvent(BaseModel): 

29 """ 

30 One system/operational log entry (logs/system/system.jsonl). 

31 

32 Inspired by: 

33 - OCSF Application Error (6008): error_type, error_message, stacktrace 

34 - OCSF Process Activity (1007): component identity, operational activity, status 

35 

36 Used for WARNING, ERROR, and CRITICAL events that indicate operational issues. 

37 

38 Note: 'time' is None when created, populated by ISO8601Formatter during logging. 

39 """ 

40 

41 # --- core --- 

42 time: Optional[str] = Field( 

43 None, 

44 description="ISO 8601 timestamp, added by formatter during serialization", 

45 ) 

46 level: str # "WARNING", "ERROR", "CRITICAL" 

47 event: str # machine-friendly event name 

48 message: Optional[str] = None # human-readable description 

49 

50 # --- component / context --- 

51 component: Optional[str] = None # "proxy", "backend_client", etc. 

52 session_id: Optional[str] = None # if tied to an MCP session 

53 request_id: Optional[str] = None # if tied to an MCP request 

54 backend_id: Optional[str] = None # if backend-specific 

55 config_version: Optional[str] = None # config active at the time 

56 

57 # --- error details --- 

58 error_type: Optional[str] = None # Exception class, e.g. TimeoutError 

59 error_message: Optional[str] = None # Short error message 

60 stacktrace: Optional[str] = None # Optional full traceback 

61 

62 # --- additional structured details --- 

63 details: Optional[Dict[str, Any]] = None # retry_count, timeout_ms, etc. 

64 

65 model_config = ConfigDict(extra="allow")

68class ConfigHistoryEvent(BaseModel): 

69 """ 

70 One configuration history log entry (logs/system/config_history.jsonl). 

71 

72 Captures the full lifecycle of configuration: 

73 - config_created: Initial configuration creation via CLI init 

74 - config_updated: Configuration updated via CLI commands 

75 - config_loaded: Configuration loaded at proxy startup 

76 - manual_change_detected: File modified outside of CLI (checksum mismatch) 

77 - config_validation_failed: Invalid JSON or schema validation error 

78 

79 The design follows general security logging and configuration 

80 management guidance (e.g., OWASP logging recommendations, 

81 NIST SP 800-128 for security-focused configuration management, 

82 and NIST SP 800-92 / CIS Control 8 for audit log management), 

83 by recording when configuration changes occur, which version 

84 is active, and a snapshot sufficient to reconstruct the effective 

85 configuration during later analysis. 

86 

87 Note: 'time' is None when created, populated by ISO8601Formatter during logging. 

88 """ 

89 

90 # --- core --- 

91 time: Optional[str] = Field( 

92 None, 

93 description="ISO 8601 timestamp, added by formatter during serialization", 

94 ) 

95 event: Literal[ 

96 "config_created", 

97 "config_updated", 

98 "config_loaded", 

99 "manual_change_detected", 

100 "config_validation_failed", 

101 ] 

102 message: Optional[str] = None # human-readable description 

103 

104 # --- versioning --- 

105 config_version: str # version ID (e.g., "v1", "v2") 

106 previous_version: Optional[str] = None 

107 change_type: Literal[ 

108 "initial_load", # First time config is created 

109 "cli_update", # CLI update command 

110 "manual_edit", # Detected manual file edit 

111 "startup_load", # Loading config on proxy startup 

112 "validation_error", # Config failed validation 

113 ] 

114 

115 # --- source / component --- 

116 component: Optional[str] = None # e.g. "cli", "proxy", "config" 

117 config_path: Optional[str] = None # path to the config file on disk 

118 source: Optional[str] = None # e.g. "cli_init", "cli_update", "proxy_startup" 

119 

120 # --- integrity --- 

121 checksum: str # e.g. "sha256:abcd1234..." 

122 

123 # --- snapshot --- 

124 snapshot_format: Literal["json"] = "json" 

125 snapshot: Optional[str] = None # full config content (optional for load events) 

126 

127 # --- change details (for update events) --- 

128 changes: Optional[Dict[str, Dict[str, Any]]] = None # {"path": {"old": x, "new": y}} 

129 

130 # --- error details (for validation failures) --- 

131 error_type: Optional[str] = None # e.g. "JSONDecodeError", "ValidationError" 

132 error_message: Optional[str] = None # human-readable error 

133 

134 model_config = ConfigDict(extra="forbid")

137class PolicyHistoryEvent(BaseModel): 

138 """ 

139 One policy history log entry (logs/system/policy_history.jsonl). 

140 

141 Captures the full lifecycle of policy configuration: 

142 - policy_created: Initial policy creation via CLI init 

143 - policy_loaded: Policy loaded at proxy startup 

144 - policy_updated: Policy updated (e.g., rule added/removed) 

145 - manual_change_detected: File modified outside of proxy (checksum mismatch) 

146 - policy_validation_failed: Invalid JSON or schema validation error 

147 

148 The design mirrors ConfigHistoryEvent for consistency. 

149 

150 Note: 'time' is None when created, populated by ISO8601Formatter during logging. 

151 """ 

152 

153 # --- core --- 

154 time: Optional[str] = Field( 

155 None, 

156 description="ISO 8601 timestamp, added by formatter during serialization", 

157 ) 

158 event: Literal[ 

159 "policy_created", 

160 "policy_loaded", 

161 "policy_updated", 

162 "manual_change_detected", 

163 "policy_validation_failed", 

164 ] 

165 message: Optional[str] = None # human-readable description 

166 

167 # --- versioning --- 

168 policy_version: str # version ID (e.g., "v1", "v2") 

169 previous_version: Optional[str] = None 

170 change_type: Literal[ 

171 "initial_creation", # First time policy is created 

172 "startup_load", # Loading policy on proxy startup 

173 "rule_update", # Rules added/removed 

174 "manual_edit", # Detected manual file edit 

175 "validation_error", # Policy failed validation 

176 ] 

177 

178 # --- source / component --- 

179 component: Optional[str] = None # e.g. "cli", "proxy", "pep", "hitl" 

180 policy_path: Optional[str] = None # path to policy.json on disk 

181 source: Optional[str] = None # e.g. "cli_init", "proxy_startup", "hitl_handler" 

182 

183 # --- integrity --- 

184 checksum: str # e.g. "sha256:abcd1234..." 

185 

186 # --- snapshot --- 

187 snapshot_format: Literal["json"] = "json" 

188 snapshot: Optional[str] = None # full policy content (optional for load events) 

189 

190 # --- rule details (for rule updates) --- 

191 rule_id: Optional[str] = None # ID of added/removed rule 

192 rule_effect: Optional[str] = None # "allow", "deny", "hitl" 

193 rule_conditions: Optional[Dict[str, Any]] = None # conditions of added rule 

194 

195 # --- error details (for validation failures) --- 

196 error_type: Optional[str] = None # e.g. "JSONDecodeError", "ValidationError" 

197 error_message: Optional[str] = None # human-readable error 

198 

199 model_config = ConfigDict(extra="forbid")
