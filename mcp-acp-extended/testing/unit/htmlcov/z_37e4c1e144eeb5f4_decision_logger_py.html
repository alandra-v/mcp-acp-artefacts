56class DecisionEventLogger: 

57 """Logs policy decision events to decisions.jsonl with fallback chain. 

58 

59 Uses fallback chain: decisions.jsonl -> system.jsonl -> emergency_audit.jsonl.

80 self._system_logger = system_logger 

81 self._backend_id = backend_id 

82 self._policy_version = policy_version

94 def log( 

95 self, 

96 decision: Decision, 

97 decision_context: DecisionContext, 

98 matched_rules: list[MatchedRule], 

99 final_rule: str, 

100 policy_eval_ms: float, 

101 hitl_outcome: HITLOutcome | None = None, 

102 policy_hitl_ms: float | None = None, 

103 hitl_cache_hit: bool | None = None, 

104 hitl_approver_id: str | None = None, 

105 ) -> None: 

106 """Log policy decision to decisions.jsonl with fallback chain. 

107 

108 Args: 

109 decision: The policy decision. 

110 decision_context: Context used for evaluation. 

111 matched_rules: Matched rules with id, effect, and description. 

112 final_rule: Rule that determined outcome. 

113 policy_eval_ms: Policy rule evaluation time. 

114 hitl_outcome: HITL outcome if applicable. 

115 policy_hitl_ms: HITL wait time if applicable. 

116 hitl_cache_hit: True if approval from cache, False if user prompted. 

117 hitl_approver_id: OIDC subject ID of user who approved/denied. 

118 

119 Raises: 

120 McpError: If primary logging fails (after logging to fallbacks). 

121 """ 

122 # Extract context summary 

123 tool = decision_context.resource.tool 

124 resource = decision_context.resource.resource 

125 

126 tool_name = tool.name if tool else None 

127 path = resource.path if resource else None 

128 source_path = resource.source_path if resource else None 

129 dest_path = resource.dest_path if resource else None 

130 uri = resource.uri if resource else None 

131 scheme = resource.scheme if resource else None 

132 

133 # Extract side effects as list of strings 

134 side_effects: list[str] | None = None 

135 if tool and tool.side_effects: 

136 side_effects = [effect.value for effect in tool.side_effects] 

137 

138 # Calculate total time (eval + HITL, excludes context) 

139 policy_total_ms = policy_eval_ms + (policy_hitl_ms or 0.0) 

140 

141 # Map USER_ALLOWED_ONCE to user_allowed for logging (same outcome, different caching) 

142 hitl_outcome_value: str | None = None 

143 if hitl_outcome: 

144 # Local import to avoid circular dependency (pep.hitl -> pep.__init__ -> pep.middleware -> here) 

145 from mcp_acp_extended.pep.hitl import HITLOutcome 

146 

147 if hitl_outcome == HITLOutcome.USER_ALLOWED_ONCE: 

148 hitl_outcome_value = "user_allowed" 

149 else: 

150 hitl_outcome_value = hitl_outcome.value 

151 

152 # Convert MatchedRule to MatchedRuleLog for logging 

153 matched_rules_log = [ 

154 MatchedRuleLog( 

155 id=rule.id, 

156 effect=rule.effect, 

157 description=rule.description, 

158 ) 

159 for rule in matched_rules 

160 ] 

161 

162 event = DecisionEvent( 

163 decision=decision.value, 

164 matched_rules=matched_rules_log, 

165 final_rule=final_rule, 

166 mcp_method=decision_context.action.mcp_method, 

167 tool_name=tool_name, 

168 path=path, 

169 source_path=source_path, 

170 dest_path=dest_path, 

171 uri=uri, 

172 scheme=scheme, 

173 subject_id=decision_context.subject.id, 

174 backend_id=self._backend_id, 

175 side_effects=side_effects, 

176 policy_version=self._policy_version or "unknown", 

177 policy_eval_ms=round(policy_eval_ms, 2), 

178 policy_hitl_ms=round(policy_hitl_ms, 2) if policy_hitl_ms else None, 

179 policy_total_ms=round(policy_total_ms, 2), 

180 request_id=decision_context.environment.request_id, 

181 session_id=decision_context.environment.session_id, 

182 hitl_outcome=hitl_outcome_value, 

183 hitl_cache_hit=hitl_cache_hit, 

184 hitl_approver_id=hitl_approver_id, 

185 ) 

186 

187 # Log with fallback chain 

188 event_data = serialize_audit_event(event) 

189 success, failure_reason = log_with_fallback( 

190 primary_logger=self._logger, 

191 system_logger=self._system_logger, 

192 event_data=event_data, 

193 event_type="decision", 

194 source_file="decisions.jsonl", 

195 ) 

196 

197 # If primary audit failed, raise error to client before shutdown 

198 if not success: 

199 raise McpError( 

200 ErrorData( 

201 code=INTERNAL_ERROR, 

202 message="Decision audit log failure - logged to fallback, proxy shutting down", 

203 )
