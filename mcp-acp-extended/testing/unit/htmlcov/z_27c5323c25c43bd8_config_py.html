145class StdioAttestationConfig(BaseModel): 

146 """Binary attestation configuration for STDIO backends. 

147 

148 Two verification modes (can use both): 

149 

150 1. SLSA Provenance (build-time): 

151 - slsa_owner: GitHub owner (user/org) for attestation verification 

152 - Proves binary was built from trusted CI/CD pipeline 

153 

154 2. Runtime checks: 

155 - expected_sha256: Verify binary hash matches expected 

156 - require_signature: Require valid code signature (macOS) 

157 

158 Attributes: 

159 slsa_owner: GitHub owner for SLSA attestation verification. 

160 If set, runs `gh attestation verify --owner <owner> <binary>`. 

161 Requires `gh` CLI to be installed and authenticated. 

162 expected_sha256: Expected SHA-256 hash of the binary (hex string). 

163 If set, binary hash is verified before spawn. 

164 require_signature: Whether to require valid code signature (macOS only). 

165 Default False (opt-in). Ignored on non-macOS platforms. 

166 """ 

167 

168 slsa_owner: str | None = None 

169 expected_sha256: str | None = None 

170 require_signature: bool = False # Opt-in for macOS codesign

218 http: Streamable HTTP transport configuration (url, timeout). 

219 """

237class HITLConfig(BaseModel): 

238 """Configuration for Human-in-the-Loop approval. 

239 

240 Attributes: 

241 timeout_seconds: How long to wait for user response (default: 60s). 

242 Must be between 5-300 seconds. 

243 default_on_timeout: What to do if user doesn't respond (always "deny"). 

244 approval_ttl_seconds: How long cached approvals remain valid (default: 600s). 

245 Must be between 300-900 seconds (5-15 minutes). 

246 cache_side_effects: Side effects that are allowed to be cached. 

247 If None (default), tools with ANY side effect are never cached. 

248 Set to a list of SideEffects to allow caching for those effects. 

249 

250 Important: 

251 The timeout should be shorter than your MCP client's request timeout. 

252 If the client times out before the user responds, the request will fail 

253 even if the user later approves. See constants.py for details. 

254 """ 

255 

256 timeout_seconds: int = Field( 

257 default=DEFAULT_HITL_TIMEOUT_SECONDS, 

258 ge=MIN_HITL_TIMEOUT_SECONDS, 

259 le=MAX_HITL_TIMEOUT_SECONDS, 

260 ) 

261 default_on_timeout: Literal["deny"] = "deny" 

262 

263 approval_ttl_seconds: int = Field( 

264 default=DEFAULT_APPROVAL_TTL_SECONDS, 

265 ge=MIN_APPROVAL_TTL_SECONDS, 

266 le=MAX_APPROVAL_TTL_SECONDS, 

267 ) 

268 cache_side_effects: list[SideEffect] | None = None

271class AppConfig(BaseModel): 

272 """Main application configuration for mcp-acp-extended. 

273 

274 Contains all configuration sections including authentication, logging, 

275 backend server, proxy settings, and HITL (Human-in-the-Loop) configuration. 

276 

277 Zero Trust: Authentication is mandatory. The proxy will not start without 

278 valid auth configuration. There is no unauthenticated fallback. 

279 

280 Attributes: 

281 auth: Authentication configuration (OIDC, mTLS). Required for proxy to start. 

282 logging: Logging configuration (log level, paths, payload settings). 

283 backend: Backend server configuration (STDIO or Streamable HTTP transport). 

284 proxy: Proxy server configuration (name). 

285 hitl: Human-in-the-Loop configuration (timeout, approval TTL, caching). 

286 """ 

287 

288 auth: AuthConfig | None = None # Validated at runtime - proxy won't start without it 

289 logging: LoggingConfig 

290 backend: BackendConfig 

291 proxy: ProxyConfig = Field(default_factory=ProxyConfig) 

292 hitl: HITLConfig = Field(default_factory=HITLConfig)

294 def save_to_file(self, config_path: Path) -> None: 

295 """Save configuration to JSON file. 

296 

297 Creates parent directories if they don't exist. 

298 Sets secure permissions (0o700) on the config directory. 

299 

300 Args: 

301 config_path: Path where mcp_acp_extended_config.json should be saved. 

302 """ 

303 config_path.parent.mkdir(parents=True, exist_ok=True) 

304 config_path.parent.chmod(0o700) 

305 

306 with open(config_path, "w", encoding="utf-8") as f: 

307 json.dump(self.model_dump(), f, indent=2) 

308 

309 config_path.chmod(0o600)

312 def load_from_files(cls, config_path: Path) -> "AppConfig": 

313 """Load configuration from JSON file. 

314

317

320 

321 Raises: 

322 FileNotFoundError: If config file doesn't exist. 

323 ValueError: If config file is invalid or missing required fields. 

324 """

331 encoding="utf-8",
