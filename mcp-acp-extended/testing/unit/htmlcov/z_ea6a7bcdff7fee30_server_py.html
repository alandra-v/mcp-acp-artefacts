1"""FastAPI server for the management API and static file serving. 

2 

3Currently implements: 

4- Cached approvals API (/api/approvals/cached) - previously approved HITL decisions 

5- Pending approvals API (/api/approvals/pending) - HITL requests waiting for decision 

6- Proxies API (/api/proxies) - proxy information 

7- Auth sessions API (/api/auth-sessions) - user authentication bindings 

8- Control API (/api/control) - policy reload 

9 

10Security: 

11- HTTP: HttpOnly cookie (production) or Bearer token (dev) for /api/* endpoints 

12- HTTP: Host header validation (DNS rebinding protection) 

13- HTTP: Origin header validation (CSRF protection) 

14- UDS: OS file permissions provide authentication (no token needed) 

15- All: Security response headers 

16 

17Future additions: 

18- Config management (/api/config) 

19- Policy management (/api/policy) 

20- Log viewer (/api/logs) 

21- Static file serving for React UI 

22 

23Usage: 

24 The API server is embedded in the proxy process (see proxy.py) to share 

25 memory with the approval store. It starts automatically on port 8765 

26 when the proxy runs. 

27 

28 For standalone development/testing (without shared memory): 

29 uv run uvicorn mcp_acp_extended.api.server:create_api_app \\ 

30 --factory --host 127.0.0.1 --port 8765 

31 

32 Note: Standalone mode won't have access to the approval cache since the 

33 ApprovalStore is only registered when running inside the proxy process. 

34 Also, security middleware is disabled in standalone mode (no token). 

35"""

76def _is_dev_mode_request(request: Request) -> bool: 

77 """Check if request is from Vite dev server (cross-origin). 

78 

79 Dev mode: Vite on :VITE_DEV_PORT makes requests to API on :8765 

80 Production: Browser loads page directly from :8765 (same-origin) 

81 

82 In dev mode, we inject the token into HTML for cross-origin auth. 

83 In production, we use HttpOnly cookies instead. 

84 """ 

85 origin = request.headers.get("origin", "") 

86 referer = request.headers.get("referer", "") 

87 dev_port_pattern = f":{VITE_DEV_PORT}" 

88 return dev_port_pattern in origin or dev_port_pattern in referer

91def create_api_app(token: str | None = None, is_uds: bool = False) -> FastAPI: 

92 """Create the FastAPI application with all routes. 

93 

94 Args: 

95 token: Bearer token for API authentication. If None, security 

96 middleware is disabled (for standalone dev/testing). 

97 is_uds: If True, this app serves UDS connections (OS permissions = auth). 

98 UDS apps skip token/host/origin validation but keep size limits 

99 and security headers. 

100 

101 Returns: 

102 Configured FastAPI application. 

103 """ 

104 app = FastAPI( 

105 title="MCP-ACP Extended API", 

106 description="Management API for MCP-ACP Extended proxy", 

107 version="0.1.0", 

108 ) 

109 

110 # Store token for injection into index.html (HTTP only, not UDS) 

111 app.state.api_token = token 

112 

113 # Mark if this is a UDS server (CLI uses UDS, browser uses HTTP) 

114 app.state.is_uds_server = is_uds 

115 

116 # Security middleware (must be added before CORS) 

117 # For UDS: skip token/host/origin checks (OS permissions = auth) 

118 # For HTTP: full validation with token 

119 # Disabled for standalone dev/testing (no token, not UDS) 

120 if token or is_uds: 

121 app.add_middleware(SecurityMiddleware, token=token, is_uds=is_uds) 

122 

123 # CORS configuration 

124 # In production, this API runs on localhost only (same-origin with proxy) 

125 # CORS is disabled by default - only enable for development with separate frontend 

126 # Set MCP_ACP_CORS_ORIGINS="http://localhost:3000" when running Vite dev server 

127 cors_origins_env = os.environ.get("MCP_ACP_CORS_ORIGINS", "").strip() 

128 if cors_origins_env: 

129 cors_origins = [origin.strip() for origin in cors_origins_env.split(",") if origin.strip()] 

130 app.add_middleware( 

131 CORSMiddleware, 

132 allow_origins=cors_origins, 

133 allow_credentials=True, 

134 allow_methods=["GET", "POST", "PUT", "DELETE", "OPTIONS"], 

135 allow_headers=["Content-Type", "Authorization"], 

136 max_age=3600, # Cache preflight for 1 hour 

137 ) 

138 

139 # Register exception handlers for structured error responses 

140 app.add_exception_handler(APIError, api_error_handler) 

141 app.add_exception_handler(RequestValidationError, validation_error_handler) 

142 app.add_exception_handler(StarletteHTTPException, http_exception_handler) 

143 

144 # Mount API routes 

145 app.include_router(proxies.router, prefix="/api/proxies", tags=["proxies"]) 

146 app.include_router(sessions.router, prefix="/api/auth-sessions", tags=["auth-sessions"]) 

147 app.include_router(approvals.router, prefix="/api/approvals/cached", tags=["cached-approvals"]) 

148 app.include_router(pending.router, prefix="/api/approvals/pending", tags=["pending-approvals"]) 

149 app.include_router(control.router, prefix="/api/control", tags=["control"]) 

150 app.include_router(policy.router, prefix="/api/policy", tags=["policy"]) 

151 app.include_router(auth.router, prefix="/api/auth", tags=["auth"]) 

152 app.include_router(config.router, prefix="/api/config", tags=["config"]) 

153 app.include_router(logs.router, prefix="/api/logs", tags=["logs"]) 

154 app.include_router(incidents.router, prefix="/api/incidents", tags=["incidents"]) 

155 app.include_router(debug.router, prefix="/api/debug", tags=["debug"]) 

156

158 # Note: We use dynamic file serving instead of StaticFiles mount so that 

159 # rebuilds are picked up without server restart (for development convenience) 

160 if STATIC_DIR.exists(): 

161 index_file = STATIC_DIR / "index.html" 

162 if index_file.exists():

164 @app.get("/{path:path}", response_model=None) 

165 async def serve_spa(path: str, request: Request) -> Response: 

166 """Serve static files or index.html for SPA routing.""" 

167 # Media type mapping for static files 

168 media_types = { 

169 ".svg": "image/svg+xml", 

170 ".ico": "image/x-icon", 

171 ".png": "image/png", 

172 ".jpg": "image/jpeg", 

173 ".jpeg": "image/jpeg", 

174 ".json": "application/json", 

175 ".js": "application/javascript", 

176 ".css": "text/css", 

177 ".woff": "font/woff", 

178 ".woff2": "font/woff2", 

179 ".ttf": "font/ttf", 

180 ".map": "application/json", 

181 } 

182 

183 # Check for static files (root-level or in assets/) 

184 if path: 

185 static_file = STATIC_DIR / path 

186 if static_file.exists() and static_file.is_file(): 

187 suffix = static_file.suffix.lower() 

188 media_type = media_types.get(suffix, "application/octet-stream") 

189 # Long cache for hashed assets, short cache for root files 

190 cache_control = ( 

191 "public, max-age=31536000, immutable" 

192 if path.startswith("assets/") 

193 else "public, max-age=3600" 

194 ) 

195 return FileResponse( 

196 static_file, 

197 media_type=media_type, 

198 headers={"Cache-Control": cache_control}, 

199 ) 

200 

201 # SPA fallback: serve index.html 

202 html = index_file.read_text() 

203 api_token = getattr(request.app.state, "api_token", None) 

204 

205 # Dev mode only: inject token into HTML for cross-origin auth 

206 # In production (same-origin), we use HttpOnly cookies instead. 

207 # Security: Token is validated to be hex-only before injection 

208 # to prevent XSS. json.dumps provides additional escaping. 

209 if _is_dev_mode_request(request) and api_token and is_valid_token_format(api_token): 

210 token_script = f"<script>window.__API_TOKEN__ = {json.dumps(api_token)};</script>" 

211 html = html.replace("</head>", f"{token_script}\n </head>") 

212 

213 response = HTMLResponse( 

214 content=html, 

215 headers={"Cache-Control": "no-cache, no-store, must-revalidate"}, 

216 ) 

217 

218 # Always set HttpOnly cookie for browser authentication 

219 # Works in production (same-origin), harmless in dev (cross-origin won't send it) 

220 if api_token: 

221 response.set_cookie( 

222 key="api_token", 

223 value=api_token, 

224 httponly=True, 

225 samesite="strict", 

226 path="/api", 

227 ) 

228 

229 return response 

230 

231 return app
