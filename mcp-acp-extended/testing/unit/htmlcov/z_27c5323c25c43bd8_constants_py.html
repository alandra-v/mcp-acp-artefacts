187def _build_transport_errors() -> tuple[type[Exception], ...]: 

188 """Build complete tuple of transport error types including httpx if available. 

189 

190 Returns: 

191 Tuple of exception types that indicate transport/connection failures. 

192 """ 

193 errors: list[type[Exception]] = list(BASE_TRANSPORT_ERRORS) 

194 

195 # Add HTTP transport errors (httpx) if available 

196 # httpx is a dependency of fastmcp's StreamableHttpTransport 

197 try: 

198 import httpx 

199 

200 # Use base classes to catch all subclasses: 

201 # - NetworkError: ConnectError, CloseError, ReadError, WriteError 

202 # - TimeoutException: ConnectTimeout, ReadTimeout, WriteTimeout, PoolTimeout 

203 # - ProtocolError: RemoteProtocolError, LocalProtocolError 

204 errors.extend( 

205 [ 

206 httpx.NetworkError, 

207 httpx.TimeoutException, 

208 httpx.ProtocolError, 

209 ] 

210 ) 

211 except ImportError: 

212 pass # httpx not available, HTTP errors won't be detected by type 

213 

214 return tuple(errors)

220# ============================================================================ 

221# HITL (Human-in-the-Loop) Configuration 

222# ============================================================================ 

223 

224# Default HITL dialog timeout (seconds) 

225# How long to wait for user to respond before auto-denying 

226# 

227# IMPORTANT: Client Timeout Considerations 

228# ---------------------------------------- 

229# MCP clients (like Claude Desktop, custom integrations) have their own 

230# request timeouts. If the client timeout is shorter than the HITL timeout, 

231# the client will timeout before the user can respond, causing the request 

232# to fail even if the user later approves. 

233# 

234# Recommendations: 

235# 1. HITL timeout should be LESS than client request timeout 

236# 2. If using HITL with long timeouts, configure clients with longer timeouts 

237# 3. Consider DEFAULT_HTTP_TIMEOUT_SECONDS (30s) when setting HITL timeout 

238# 

239# Example conflict scenarios: 

240# - Client timeout: 30s, HITL timeout: 30s → Client times out during approval 

241# - Client timeout: 60s, HITL timeout: 30s → Safe, 30s buffer for user response 

242# 

243DEFAULT_HITL_TIMEOUT_SECONDS: int = 60 

244 

245# HITL timeout validation range (seconds) 

246MIN_HITL_TIMEOUT_SECONDS: int = 5 # Minimum time for user to read and respond 

247MAX_HITL_TIMEOUT_SECONDS: int = 300 # 5 minutes max
