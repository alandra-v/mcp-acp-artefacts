83def _needs_normalization(raw_rules: list[dict[str, object]], validated_rules: list[PolicyRule]) -> bool: 

84 """Check if any rule IDs were auto-generated during validation. 

85 

86 Compares raw JSON rules against validated PolicyRule objects to detect 

87 if the ensure_rule_ids validator generated any new IDs. 

88 

89 Args: 

90 raw_rules: Original rules from JSON file (may have None/missing IDs). 

91 validated_rules: Validated PolicyRule objects (always have IDs). 

92 

93 Returns: 

94 True if any IDs were generated and file should be updated. 

95 """ 

96 if len(raw_rules) != len(validated_rules): 

97 return False # Length mismatch shouldn't happen, be defensive 

98 

99 for i, raw_rule in enumerate(raw_rules): 

100 # If raw rule had no ID but validated rule has one, it was generated 

101 if raw_rule.get("id") is None and validated_rules[i].id is not None: 

102 return True 

103 

104 return False

107def load_policy(path: Path | None = None, *, normalize: bool = True) -> PolicyConfig: 

108 """Load policy configuration from file. 

109 

110 Automatically normalizes the policy file by generating missing rule IDs. 

111 This keeps the file in sync with the runtime representation. 

112 

113 Args: 

114 path: Path to policy.json. If None, uses default location. 

115 normalize: If True (default), save back to file if IDs were generated. 

116 Set to False to skip normalization (e.g., for read-only checks). 

117 

118 Returns: 

119 PolicyConfig loaded from file. 

120 

121 Raises: 

122 FileNotFoundError: If policy file does not exist. 

123 ValueError: If policy file contains invalid JSON or schema. 

124 

125 Note: 

126 If normalization save fails (e.g., permission error), a warning is logged 

127 but the valid policy is still returned. The file will be normalized on 

128 the next successful save. 

129 """ 

130 policy_path = path or get_policy_path() 

131 require_file_exists(policy_path, file_type="policy") 

132 

133 # Load raw JSON (single read) 

134 try: 

135 with open(policy_path, encoding="utf-8") as f: 

136 raw_data = json.load(f) 

137 except json.JSONDecodeError as e: 

138 raise ValueError(f"Invalid JSON in policy file {policy_path}: {e}") from e 

139 

140 # Validate (triggers ensure_rule_ids which generates missing IDs) 

141 try: 

142 policy = PolicyConfig.model_validate(raw_data) 

143 except ValidationError as e: 

144 errors = [] 

145 for error in e.errors(): 

146 loc = ".".join(str(x) for x in error["loc"]) 

147 errors.append(f" - {loc}: {error['msg']}") 

148 raise ValueError( 

149 f"Invalid policy configuration in {policy_path}:\n" 

150 + "\n".join(errors) 

151 + "\n\nEdit the policy file or run 'mcp-acp-extended init' to recreate." 

152 ) from e 

153 

154 # Auto-normalize: save back if IDs were generated 

155 if normalize: 

156 raw_rules = raw_data.get("rules", []) 

157 if _needs_normalization(raw_rules, policy.rules): 

158 try: 

159 save_policy(policy, policy_path) 

160 except OSError as e: 

161 # Log warning but don't fail - policy is valid, just not persisted 

162 logger = logging.getLogger(__name__) 

163 logger.warning( 

164 "Failed to save normalized policy to %s: %s. " 

165 "Generated rule IDs will not be persisted until next successful save.", 

166 policy_path, 

167 e, 

168 ) 

169 

170 return policy

173def save_policy(policy: PolicyConfig, path: Path | None = None) -> None: 

174 """Save policy configuration to file atomically. 

175 

176 Uses atomic write pattern: write to temp file, then rename. 

177 This prevents file corruption if write fails midway. 

178 

179 Creates parent directories if they don't exist. 

180 Sets secure permissions (0o700 on directory, 0o600 on file). 

181 

182 Args: 

183 policy: PolicyConfig to save. 

184 path: Path to save to. If None, uses default location. 

185 """ 

186 policy_path = path or get_policy_path() 

187 

188 # Ensure parent directory exists with secure permissions 

189 policy_path.parent.mkdir(parents=True, exist_ok=True) 

190 set_secure_permissions(policy_path.parent, is_directory=True) 

191 

192 # Convert to dict and format as JSON 

193 data = policy.model_dump(mode="json") 

194 content = json.dumps(data, indent=2) + "\n" # Trailing newline 

195 

196 # Atomic write: write to temp file in same directory, then rename 

197 # Same directory ensures rename is atomic (same filesystem) 

198 fd, temp_path = tempfile.mkstemp( 

199 dir=policy_path.parent, 

200 prefix=".policy_", 

201 suffix=".tmp", 

202 ) 

203 try: 

204 # Write content to temp file 

205 with os.fdopen(fd, "w") as f: 

206 f.write(content) 

207 f.flush() 

208 os.fsync(f.fileno()) # Ensure data is on disk 

209 

210 # Set secure permissions on temp file before rename 

211 os.chmod(temp_path, 0o600) 

212 

213 # Atomic rename (overwrites existing file atomically) 

214 os.replace(temp_path, policy_path) 

215 

216 except Exception: 

217 # Clean up temp file on failure 

218 try: 

219 os.unlink(temp_path) 

220 except OSError: 

221 pass 

222 raise
