1"""Policy engine - evaluate DecisionContext against policy rules. 

2 

3This module provides the PolicyEngine class that evaluates requests 

4against policy rules to produce ALLOW/DENY/HITL decisions. 

5 

6Evaluation flow: 

71. Discovery methods (tools/list, etc.) → ALLOW (bypass policy) 

82. Collect all matching rules 

93. Apply combining algorithm: HITL > DENY > ALLOW 

104. Within each effect level, select the most specific rule 

115. No match → return default_action (DENY) 

12 

13Design principles: 

141. All conditions in a rule use AND logic 

152. Combining algorithm: HITL overrides DENY overrides ALLOW 

163. Within the same effect level, the most specific rule wins 

174. Discovery methods bypass policy entirely 

185. Default to DENY if no rule matches (zero trust) 

19 

20Specificity scoring: 

21Rules are ranked by specificity to determine the "final_rule" when multiple 

22rules with the same effect match. Specificity is calculated as: 

23 

24 Score = (condition_count × 100) + exactness_bonus + path_depth_bonus 

25 

26Where: 

27- condition_count: Number of non-null conditions (+100 each) 

28- exactness_bonus: +10 for each pattern without wildcards (*, ?, **) 

29- path_depth_bonus: +1 per path segment before wildcard in path patterns 

30 

31Example scores: 

32- tool_name: "read*" → 100 (1 condition) 

33- tool_name: "read_file" → 110 (1 condition + exact) 

34- tool_name: "read*", extension: ".py" → 200 (2 conditions) 

35- tool_name: "read*", path_pattern: "/a/b/c/**" → 203 (2 conditions + depth 3) 

36 

37Tie-breaker: If two rules have the same specificity score, the rule 

38that appears first in the policy file wins (preserves predictability). 

39 

40Known limitations: 

41- Bracket glob patterns `[...]` are not detected as wildcards 

42- Escaped wildcards are not supported 

43- List conditions count as 1 condition regardless of list length 

44"""

78def _has_wildcards(pattern: str | list[str] | None) -> bool: 

79 """Check if a pattern contains wildcard characters (*, ?). 

80 

81 For lists, returns True if ANY pattern has wildcards. 

82 

83 Args: 

84 pattern: Single pattern string, list of patterns, or None. 

85 

86 Returns: 

87 True if any wildcard character is present, False otherwise. 

88 """ 

89 if pattern is None: 

90 return False 

91 if isinstance(pattern, str): 

92 return any(c in pattern for c in _WILDCARD_CHARS) 

93 # List of patterns - check if any has wildcards 

94 return any(any(c in p for c in _WILDCARD_CHARS) for p in pattern)

97def _count_path_depth(pattern: str | list[str] | None) -> int: 

98 """Count path segments before the first wildcard. 

99 

100 Deeper paths are more specific (e.g., /a/b/c/** scores higher than /a/**). 

101 For lists, returns the maximum depth across all patterns. 

102 

103 Args: 

104 pattern: Path pattern string, list of patterns, or None. 

105 

106 Returns: 

107 Number of path segments before the first wildcard. 

108 """ 

109 if pattern is None: 

110 return 0 

111 

112 def _depth_for_single(p: str) -> int: 

113 # Split by / and count segments before first wildcard 

114 segments = p.split("/") 

115 depth = 0 

116 for seg in segments: 

117 if any(c in seg for c in _WILDCARD_CHARS): 

118 break 

119 if seg: # Skip empty segments (leading /) 

120 depth += 1 

121 return depth 

122 

123 if isinstance(pattern, str): 

124 return _depth_for_single(pattern) 

125 # List of patterns - return max depth 

126 return max(_depth_for_single(p) for p in pattern) if pattern else 0

147class PolicyEngine: 

148 """Policy evaluation engine. 

149 

150 Evaluates DecisionContext against policy rules to produce decisions. 

151 Discovery methods bypass policy entirely and always return ALLOW. 

152 

153 Uses HITL > DENY > ALLOW combining algorithm: 

154 - If ANY matching rule says HITL → HITL (human decides) 

155 - Else if ANY matching rule says DENY → DENY 

156 - Else if ANY matching rule says ALLOW → ALLOW 

157 - Else → default_action (DENY) 

158 

159 Attributes: 

160 policy: The policy configuration to evaluate against. 

161 """ 

162

164 self, 

165 policy: PolicyConfig, 

166 protected_dirs: tuple[str, ...] = (), 

167 ) -> None:

169 

170 Args: 

171 policy: Policy configuration with rules and defaults.

192 def is_protected_path(self, path: str | None) -> bool: 

193 """Check if path is under a protected directory. 

194 

195 Protected paths cannot be accessed by MCP tools regardless of policy. 

196 This is a built-in security measure that cannot be overridden. 

197 

198 Symlink-safe: resolves path to real path before checking. 

199 

200 Args: 

201 path: File path to check (may be None). 

202 

203 Returns: 

204 True if path is under a protected directory, False otherwise. 

205 """ 

206 if path is None or not self._protected_dirs: 

207 return False 

208 

209 try: 

210 resolved = os.path.realpath(path) 

211 return any(resolved == d or resolved.startswith(d + os.sep) for d in self._protected_dirs) 

212 except (OSError, ValueError): 

213 # Can't resolve path - not a protected path match 

214 return False

216 def evaluate(self, context: DecisionContext) -> Decision: 

217 """Evaluate a request context against the policy. 

218 

219 Evaluation order: 

220 1. Built-in protected paths → DENY (cannot be overridden) 

221 2. Discovery methods → ALLOW (bypass policy) 

222 3. Collect matching rules → apply HITL > DENY > ALLOW combining 

223 4. No match → default_action (DENY) 

224 

225 Args: 

226 context: DecisionContext built from the request. 

227 

228 Returns: 

229 Decision: ALLOW, DENY, or HITL based on policy evaluation. 

230 

231 Raises: 

232 PolicyEnforcementFailure: If policy evaluation fails unexpectedly. 

233 This is a critical security failure requiring proxy shutdown. 

234 """ 

235 try: 

236 # Built-in protection - check FIRST, cannot be overridden by policy 

237 path = context.resource.resource.path if context.resource.resource else None 

238 if self.is_protected_path(path): 

239 return Decision.DENY 

240 

241 # Discovery methods bypass policy entirely 

242 if context.action.category == ActionCategory.DISCOVERY: 

243 return Decision.ALLOW 

244 

245 # Collect all matching rules 

246 matching_rules = [rule for rule in self.policy.rules if self._rule_matches(rule, context)] 

247 

248 if not matching_rules: 

249 # No rules matched - return default action (DENY) 

250 return self._effect_to_decision(self.policy.default_action) 

251 

252 # Apply combining algorithm: HITL > DENY > ALLOW 

253 # HITL has highest priority - let human decide 

254 for rule in matching_rules: 

255 if rule.effect == "hitl": 

256 return Decision.HITL 

257 

258 # DENY has second priority - security first 

259 for rule in matching_rules: 

260 if rule.effect == "deny": 

261 return Decision.DENY 

262 

263 # All matching rules must be ALLOW 

264 return Decision.ALLOW 

265 

266 except PolicyEnforcementFailure: 

267 # Re-raise our own exceptions 

268 raise 

269 except Exception as e: 

270 # Unexpected error during policy evaluation - critical failure 

271 # Cannot trust policy decisions if evaluation crashes 

272 raise PolicyEnforcementFailure( 

273 f"Policy evaluation failed unexpectedly: {type(e).__name__}: {e}. " 

274 "Cannot safely evaluate requests - proxy must shutdown." 

275 ) from e

277 def get_matching_rules(self, context: DecisionContext) -> list[MatchedRule]: 

278 """Get all rules that match the given context. 

279 

280 This is the public API for retrieving matched rules. Use this instead 

281 of accessing _rule_matches directly. 

282 

283 Each matched rule includes a specificity score for tie-breaking when 

284 multiple rules with the same effect match. Higher scores indicate 

285 more specific rules. 

286 

287 Args: 

288 context: DecisionContext to match against. 

289 

290 Returns: 

291 List of MatchedRule with id, description, effect, and specificity 

292 for each matching rule. 

293 """ 

294 matched: list[MatchedRule] = [] 

295 

296 for idx, rule in enumerate(self.policy.rules): 

297 if self._rule_matches(rule, context): 

298 rule_id = rule.id or f"rule_{idx}" 

299 matched.append( 

300 MatchedRule( 

301 id=rule_id, 

302 description=rule.description, 

303 effect=rule.effect, 

304 specificity=self._calculate_specificity(rule), 

305 ) 

306 ) 

307 

308 return matched

310 def _rule_matches(self, rule: PolicyRule, context: DecisionContext) -> bool: 

311 """Check if a rule matches the context. 

312 

313 All conditions use AND logic - all specified conditions must match. 

314 

315 Args: 

316 rule: Policy rule to check. 

317 context: DecisionContext to match against. 

318 

319 Returns: 

320 True if all conditions match, False otherwise. 

321 """ 

322 conditions = rule.conditions 

323 

324 # Extract values from context for easy access 

325 tool_name = context.resource.tool.name if context.resource.tool else None 

326 tool_side_effects = context.resource.tool.side_effects if context.resource.tool else None 

327 path = context.resource.resource.path if context.resource.resource else None 

328 source_path = context.resource.resource.source_path if context.resource.resource else None 

329 dest_path = context.resource.resource.dest_path if context.resource.resource else None 

330 extension = context.resource.resource.extension if context.resource.resource else None 

331 scheme = context.resource.resource.scheme if context.resource.resource else None 

332 backend_id = context.resource.server.id 

333 resource_type = context.resource.type.value # Convert enum to string 

334 mcp_method = context.action.mcp_method 

335 subject_id = context.subject.id 

336 

337 # Check tool_name condition (glob, case-insensitive, OR logic for lists) 

338 if not _match_any(conditions.tool_name, tool_name, match_tool_name): 

339 return False 

340 

341 # Check path_pattern condition (glob with **, OR logic for lists) 

342 if not _match_any(conditions.path_pattern, path, match_path_pattern): 

343 return False 

344 

345 # Check source_path condition (glob with **, OR logic for lists) 

346 if not _match_any(conditions.source_path, source_path, match_path_pattern): 

347 return False 

348 

349 # Check dest_path condition (glob with **, OR logic for lists) 

350 if not _match_any(conditions.dest_path, dest_path, match_path_pattern): 

351 return False 

352 

353 # Check operations condition (heuristic from tool name) 

354 if conditions.operations is not None: 

355 inferred_op = infer_operation(tool_name) 

356 if not _match_operations(conditions.operations, inferred_op): 

357 return False 

358 

359 # Check extension condition (exact, case-insensitive, OR logic for lists) 

360 if not _match_any(conditions.extension, extension, _match_exact_case_insensitive): 

361 return False 

362 

363 # Check scheme condition (exact, case-insensitive, OR logic for lists) 

364 if not _match_any(conditions.scheme, scheme, _match_exact_case_insensitive): 

365 return False 

366 

367 # Check backend_id condition (glob, case-insensitive, OR logic for lists) 

368 if not _match_any(conditions.backend_id, backend_id, _match_glob_case_insensitive): 

369 return False 

370 

371 # Check resource_type condition (exact, case-insensitive) - single value only 

372 if conditions.resource_type is not None: 

373 if not _match_exact_case_insensitive(conditions.resource_type, resource_type): 

374 return False 

375 

376 # Check mcp_method condition (glob, case-sensitive, OR logic for lists) 

377 if not _match_any(conditions.mcp_method, mcp_method, _match_glob_case_sensitive): 

378 return False 

379 

380 # Check subject_id condition (exact, case-sensitive, OR logic for lists) 

381 if not _match_any(conditions.subject_id, subject_id, _match_exact_case_sensitive): 

382 return False 

383 

384 # Check side_effects condition (ANY logic) 

385 if conditions.side_effects is not None: 

386 if not _match_side_effects(conditions.side_effects, tool_side_effects): 

387 return False 

388 

389 # All specified conditions matched 

390 return True

392 def _calculate_specificity(self, rule: PolicyRule) -> int: 

393 """Calculate specificity score for a rule. 

394 

395 Higher scores indicate more specific rules. Used to select the 

396 "final_rule" when multiple rules with the same effect match. 

397 

398 Scoring formula: 

399 Score = (condition_count × 100) + exactness_bonus + path_depth_bonus 

400 

401 Where: 

402 - condition_count: Number of non-null conditions (+100 each) 

403 - exactness_bonus: +10 for patterns without wildcards (*, ?, **) 

404 - path_depth_bonus: +1 per path segment before wildcard 

405 

406 Args: 

407 rule: PolicyRule to calculate specificity for. 

408 

409 Returns: 

410 Integer specificity score (higher = more specific). 

411 """ 

412 score = 0 

413 conditions = rule.conditions 

414 

415 # Count conditions and add exactness bonuses 

416 # Each non-null condition adds 100 points 

417 # Exact patterns (no wildcards) add 10 bonus points 

418 

419 if conditions.tool_name is not None: 

420 score += 100 

421 if not _has_wildcards(conditions.tool_name): 

422 score += 10 

423 

424 if conditions.path_pattern is not None: 

425 score += 100 

426 if not _has_wildcards(conditions.path_pattern): 

427 score += 10 

428 # Path depth bonus: deeper paths are more specific 

429 score += _count_path_depth(conditions.path_pattern) 

430 

431 if conditions.source_path is not None: 

432 score += 100 

433 if not _has_wildcards(conditions.source_path): 

434 score += 10 

435 score += _count_path_depth(conditions.source_path) 

436 

437 if conditions.dest_path is not None: 

438 score += 100 

439 if not _has_wildcards(conditions.dest_path): 

440 score += 10 

441 score += _count_path_depth(conditions.dest_path) 

442 

443 if conditions.operations is not None: 

444 score += 100 

445 

446 if conditions.extension is not None: 

447 score += 100 

448 # Extensions are typically exact (no wildcards) 

449 

450 if conditions.scheme is not None: 

451 score += 100 

452 

453 if conditions.backend_id is not None: 

454 score += 100 

455 if not _has_wildcards(conditions.backend_id): 

456 score += 10 

457 

458 if conditions.resource_type is not None: 

459 score += 100 

460 # resource_type is always exact (enum value) 

461 score += 10 

462 

463 if conditions.mcp_method is not None: 

464 score += 100 

465 if not _has_wildcards(conditions.mcp_method): 

466 score += 10 

467 

468 if conditions.subject_id is not None: 

469 score += 100 

470 # subject_id is always exact (no wildcards supported) 

471 score += 10 

472 

473 if conditions.side_effects is not None: 

474 score += 100 

475 

476 return score

478 def _get_most_specific_rule(self, rules: list[PolicyRule]) -> PolicyRule: 

479 """Get the most specific rule from a list of rules. 

480 

481 Uses specificity scoring to determine which rule is most specific. 

482 If two rules have the same score, the first one in the list wins 

483 (preserves policy file order for predictability). 

484 

485 Args: 

486 rules: Non-empty list of PolicyRule objects. 

487 

488 Returns: 

489 The most specific rule from the list. 

490 """ 

491 # Sort by specificity (descending), preserving original order for ties 

492 # enumerate() captures original index for stable sorting 

493 scored = [(self._calculate_specificity(rule), idx, rule) for idx, rule in enumerate(rules)] 

494 # Sort by score descending, then by original index ascending (for ties) 

495 scored.sort(key=lambda x: (-x[0], x[1])) 

496 return scored[0][2]
