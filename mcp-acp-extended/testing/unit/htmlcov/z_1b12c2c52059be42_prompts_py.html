34def prompt_with_retry(prompt_text: str) -> str: 

35 """Prompt for a required value, retrying if empty. 

36 

37 Args: 

38 prompt_text: Text to show in prompt. 

39 

40 Returns: 

41 Non-empty string value from user. 

42 """ 

43 while True: 

44 value: str = click.prompt(prompt_text, type=str, default="", show_default=False) 

45 if value.strip(): 

46 return value.strip() 

47 click.echo(" This field is required.")

64def prompt_stdio_config() -> StdioTransportConfig: 

65 """Prompt for STDIO transport configuration. 

66 

67 Returns: 

68 StdioTransportConfig with user-provided values. 

69 """ 

70 click.echo("\n[STDIO Configuration]") 

71 command = prompt_with_retry("Command to run") 

72 args_str = prompt_with_retry("Arguments (comma-separated)") 

73 args_list = [arg.strip() for arg in args_str.split(",") if arg.strip()] 

74 

75 # Optional attestation configuration 

76 attestation_config = prompt_stdio_attestation_config() 

77 

78 return StdioTransportConfig(command=command, args=args_list, attestation=attestation_config)

81def prompt_stdio_attestation_config() -> StdioAttestationConfig | None: 

82 """Prompt for STDIO binary attestation configuration. 

83 

84 Returns: 

85 StdioAttestationConfig if user configures attestation, None otherwise. 

86 """ 

87 click.echo("\n--- Binary Attestation (Optional) ---") 

88 click.echo("Attestation verifies the backend binary before spawning.") 

89 click.echo("Options:") 

90 click.echo(" - SLSA Provenance: Verify build attestation via GitHub CLI") 

91 click.echo(" - SHA-256 Hash: Verify binary matches expected hash") 

92 click.echo(" - Code Signature: Require valid signature (macOS only)") 

93 click.echo() 

94 

95 if not click.confirm("Configure binary attestation?", default=False): 

96 return None 

97 

98 click.echo() 

99 

100 # SLSA owner 

101 click.echo("SLSA Provenance (optional):") 

102 click.echo(" GitHub owner (user/org) that built the binary.") 

103 click.echo(" Requires `gh` CLI to be installed and authenticated.") 

104 slsa_owner = prompt_optional(" SLSA owner (leave empty to skip)") 

105 slsa_owner = slsa_owner if slsa_owner else None 

106 

107 # Expected SHA-256 

108 click.echo("\nSHA-256 Hash (optional):") 

109 click.echo(" Expected hash of the binary (hex string).") 

110 click.echo(" Get it with: shasum -a 256 /path/to/binary") 

111 expected_sha256: str | None = None 

112 while True: 

113 sha_input = prompt_optional(" Expected SHA-256 (leave empty to skip)") 

114 if not sha_input: 

115 break 

116 is_valid, normalized = validate_sha256_hex(sha_input) 

117 if is_valid: 

118 expected_sha256 = normalized 

119 break 

120 click.echo(" Invalid SHA-256 hash. Must be 64 hex characters.") 

121 

122 # Code signature (macOS only) 

123 require_signature = False 

124 if platform.system() == "Darwin": 

125 click.echo("\nCode Signature (macOS):") 

126 click.echo(" Require valid code signature from Apple or identified developer.") 

127 require_signature = click.confirm(" Require code signature?", default=False) 

128 

129 # Only return config if at least one option is configured 

130 if not slsa_owner and not expected_sha256 and not require_signature: 

131 click.echo("\nNo attestation options configured, skipping.") 

132 return None 

133 

134 return StdioAttestationConfig( 

135 slsa_owner=slsa_owner, 

136 expected_sha256=expected_sha256, 

137 require_signature=require_signature, 

138 )

141def prompt_http_config() -> HttpTransportConfig: 

142 """Prompt for HTTP transport configuration and test connectivity. 

143 

144 Returns: 

145 HttpTransportConfig with user-provided values. 

146 

147 Raises: 

148 click.Abort: If user aborts after connection failure. 

149 """ 

150 click.echo("\n[HTTP Configuration]") 

151 

152 while True: 

153 url = prompt_with_retry("Server URL (e.g., https://host:port/mcp)") 

154 timeout_str = prompt_optional("Connection timeout (seconds)", str(DEFAULT_HTTP_TIMEOUT_SECONDS)) 

155 try: 

156 timeout = int(timeout_str) 

157 except ValueError: 

158 click.echo(f" Invalid timeout, using default {DEFAULT_HTTP_TIMEOUT_SECONDS}") 

159 timeout = DEFAULT_HTTP_TIMEOUT_SECONDS 

160 

161 # Test connection 

162 click.echo(f"\nTesting connection to {url}...") 

163 try: 

164 check_http_health(url, timeout=min(timeout, HEALTH_CHECK_TIMEOUT_SECONDS)) 

165 click.echo("Server is reachable!") 

166 return HttpTransportConfig(url=url, timeout=timeout) 

167 except Exception as e: 

168 error_str = str(e).lower() 

169 click.echo(f"\nHealth check failed: could not reach {url}") 

170 click.echo(f" Error: {e}") 

171 if "ssl" in error_str or "certificate" in error_str: 

172 click.echo("\n Note: SSL/TLS error detected. If this server requires mTLS,") 

173 click.echo(" you can continue - mTLS will be configured in the next step.") 

174 click.echo("What would you like to do?") 

175 click.echo(" 1. Continue - server may be offline temporarily") 

176 click.echo(" 2. Reconfigure - enter a different URL") 

177 click.echo(" 3. Cancel - abort setup") 

178 choice = click.prompt("Select an option", type=click.IntRange(1, 3), default=1) 

179 if choice == 1: 

180 return HttpTransportConfig(url=url, timeout=timeout) 

181 elif choice == 3: 

182 raise click.Abort() 

183 # choice == 2: loop continues

186def prompt_auth_config(http_config: HttpTransportConfig | None) -> AuthConfig: 

187 """Prompt for authentication configuration. 

188 

189 Guides user through OIDC and mTLS configuration with validation. 

190 

191 Args: 

192 http_config: HTTP config if using HTTP transport. mTLS prompts 

193 are only shown for HTTPS backends. 

194 

195 Returns: 

196 AuthConfig with user-provided values. 

197 """ 

198 click.echo("\n--- Authentication ---") 

199 click.echo("Configure Auth0/OIDC for user authentication.\n") 

200 

201 # OIDC settings 

202 issuer = prompt_with_retry("OIDC issuer URL (e.g., https://your-tenant.auth0.com)") 

203 client_id = prompt_with_retry("Auth0 client ID") 

204 audience = prompt_with_retry("API audience (e.g., https://your-api.example.com)") 

205 

206 oidc_config = OIDCConfig( 

207 issuer=issuer, 

208 client_id=client_id, 

209 audience=audience, 

210 ) 

211 

212 # mTLS settings (only for HTTPS backends) 

213 mtls_config: MTLSConfig | None = None 

214 if http_config and http_config.url.startswith("https://"): 

215 click.echo("\n--- mTLS (Mutual TLS) ---") 

216 click.echo("HTTPS backend detected. mTLS allows the proxy to authenticate") 

217 click.echo("itself to the backend using a client certificate.\n") 

218 click.echo("You need 3 PEM files (obtain from your security team):") 

219 click.echo(" - Client certificate: proves proxy identity to backend") 

220 click.echo(" - Client private key: must match certificate (keep secure)") 

221 click.echo(" - CA bundle: verifies backend server's certificate\n") 

222 click.echo("Skip if your backend doesn't require client certificates.\n") 

223 

224 if click.confirm("Configure mTLS?", default=False): 

225 click.echo("\n (Paths support ~ expansion, e.g., ~/.mcp-certs/client.pem)") 

226 

227 while True: 

228 client_cert = prompt_with_retry("Client certificate path") 

229 client_key = prompt_with_retry("Client private key path") 

230 ca_bundle = prompt_with_retry("CA bundle path") 

231 

232 # Validate certificates 

233 click.echo("\nValidating certificates...") 

234 errors = validate_mtls_config(client_cert, client_key, ca_bundle) 

235 

236 if not errors: 

237 click.echo(click.style(" Certificates valid!", fg="green")) 

238 mtls_config = MTLSConfig( 

239 client_cert_path=client_cert, 

240 client_key_path=client_key, 

241 ca_bundle_path=ca_bundle, 

242 ) 

243 break 

244 

245 # Show errors 

246 click.echo(click.style("\n Certificate validation failed:", fg="red")) 

247 for error in errors: 

248 click.echo(f" - {error}") 

249 click.echo() 

250 

251 # Ask what to do 

252 choice = click.prompt( 

253 "What would you like to do?", 

254 type=click.Choice(["retry", "skip", "continue"]), 

255 default="retry", 

256 show_choices=True, 

257 ) 

258 

259 if choice == "skip": 

260 click.echo(" Skipping mTLS configuration.") 

261 break 

262 elif choice == "continue": 

263 click.echo(" Saving config with invalid certificates (will fail at startup).") 

264 mtls_config = MTLSConfig( 

265 client_cert_path=client_cert, 

266 client_key_path=client_key, 

267 ca_bundle_path=ca_bundle, 

268 ) 

269 break 

270 # else retry - loop continues 

271 

272 click.echo("\nNote: Device health (disk encryption, firewall) is checked at startup.") 

273 

274 return AuthConfig( 

275 oidc=oidc_config, 

276 mtls=mtls_config, 

277 )
