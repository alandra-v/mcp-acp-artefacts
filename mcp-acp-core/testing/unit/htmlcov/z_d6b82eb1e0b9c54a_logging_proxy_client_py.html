1"""LoggingProxyClient wrapper for Proxy↔Backend wire-level debugging. 

2 

3This module provides wire-level DEBUG logging for all communication between 

4the proxy and backend MCP servers. 

5 

6Wire logs include: 

7- Method names and metadata (tool names, file paths, operation types) 

8- Response summaries and metrics 

9- Errors and duration data 

10- Server information during initialization 

11 

12Logs are written to <log_dir>/mcp_acp_core_logs/debug/backend_wire.jsonl using Pydantic models. 

13The log_dir is specified in the user's configuration file. 

14 

15IMPORTANT - Initialize Logging Limitation: 

16Initialize is logged as INGRESS ONLY (backend_response), with no corresponding 

17EGRESS (proxy_request). This is because FastMCP's ProxyClient handles the MCP 

18initialize handshake internally within __aenter__(), and we cannot intercept 

19the outgoing request - only the completed result afterwards. This is documented 

20behavior and correctly reflects what we can observe from our position in the 

21call stack. 

22"""

50class LoggingProxyClient(ProxyClient): 

51 """ProxyClient wrapper for wire-level debug logging of backend communication. 

52 

53 Logs parsed method calls with metadata and summaries. All logs use Pydantic models: 

54 - ProxyRequestEvent: Proxy request to backend (egress) 

55 - BackendResponseEvent: Backend response to proxy (ingress, success) 

56 - BackendErrorEvent: Backend error to proxy (ingress, error) 

57 

58 Security metadata extraction (NO file content logging): 

59 - Tool names and operation types (read/write/delete/list) 

60 - File paths (normalized) 

61 - File sizes and extensions 

62 - NO file contents or hashes 

63 """ 

64 

65 # Class-level set to track logged sessions across all instances. 

66 # This prevents duplicate initialize logs when multiple client instances 

67 # (created via .new()) exist for the same session. We track by session_id 

68 # because FastMCP creates new initialize_result objects for each instance, 

69 # making object-ID-based tracking ineffective. 

70 _logged_sessions_global: set[str] = set() 

71 

72 def __init__( 

73 self, 

74 transport_or_client: ClientTransport | ProxyClient, 

75 logger: logging.Logger, 

76 transport_type: str = "stdio", 

77 system_logger: Optional[logging.Logger] = None, 

78 _is_wrapper: bool = False, 

79 ) -> None: 

80 """Initialize the logging wrapper. 

81 

82 Args: 

83 transport_or_client: Transport (initial) or ProxyClient (from .new()) 

84 logger: Logger instance for writing backend wire logs. 

85 transport_type: Transport type being used ("stdio" or "http"). 

86 system_logger: Logger instance for system/operational errors. If None, uses singleton. 

87 _is_wrapper: Internal flag - True when wrapping a client from .new() 

88 """

90 self._system_logger = system_logger or get_system_logger() 

91 self._transport = transport_type 

92 self._is_wrapper = _is_wrapper 

93 

94 if _is_wrapper: 

95 # Wrapping mode: wrapping a ProxyClient from .new() 

96 self._client = transport_or_client 

97 else: 

98 # Initial mode: inherit from ProxyClient with transport 

99 super().__init__(transport_or_client) 

100 self._client = None

102 async def _wrap_call( 

103 self, 

104 method: str, 

105 wrapped_fn: Callable[..., Any], 

106 *args: Any, 

107 **log_kwargs: Any, 

108 ) -> Any: 

109 """Wrap an async method call with logging. 

110 

111 Args: 

112 method: MCP method name 

113 wrapped_fn: The actual async function to call 

114 *args: Positional arguments to pass to wrapped_fn 

115 **log_kwargs: Additional keyword arguments for request logging 

116 

117 Returns: 

118 The result from wrapped_fn 

119 

120 Raises: 

121 Exception: Re-raises any exception from wrapped_fn after logging 

122 """ 

123 start_time = log_proxy_request( 

124 self._logger, self._transport, method, self._system_logger, **log_kwargs 

125 ) 

126 try: 

127 result = await wrapped_fn(*args) 

128 log_backend_response(self._logger, self._transport, method, start_time, result) 

129 return result 

130 except Exception as e: 

131 log_backend_error(self._logger, self._transport, method, start_time, e) 

132 raise

275 async def __aenter__(self) -> "LoggingProxyClient": 

276 """Async context manager entry - log initialize handshake. 

277 

278 IMPORTANT: Why we log initialize here (retroactively) instead of "when it happens": 

279 

280 FastMCP's ProxyClient handles the MCP initialize handshake INTERNALLY during 

281 __aenter__() - it's not exposed as a method we can wrap like call_tool() or 

282 list_tools(). The initialize happens inside self._wrapped_client.__aenter__() 

283 and we only get access to the result afterwards via initialize_result attribute. 

284 

285 This means we must log initialize retroactively after the connection is established, 

286 rather than intercepting the actual request/response like we do for other methods. 

287 

288 Why we use class-level session tracking (_logged_sessions_global): 

289 

290 FastMCP uses .new() to create multiple client instances for session isolation. 

291 Each instance gets its own initialize_result object (different object IDs), but 

292 they all belong to the same session (same session_id). Without global tracking 

293 by session_id, each instance would log initialize separately, creating duplicates. 

294 

295 By tracking session_id globally, we ensure initialize is logged exactly once per 

296 session, regardless of how many client instances are created. 

297 

298 Returns: 

299 LoggingProxyClient: The logging client instance. 

300 

301 Raises: 

302 Exception: If connection fails. 

303 """ 

304 # Enter context manager 

305 if self._is_wrapper: 

306 await self._client.__aenter__() # type: ignore[union-attr] 

307 else: 

308 await super().__aenter__() 

309 

310 # Log initialize response (once per session, globally) 

311 try: 

312 init_result_obj = self._client if self._is_wrapper else self 

313 if hasattr(init_result_obj, "initialize_result") and init_result_obj.initialize_result: 

314 init_result = init_result_obj.initialize_result 

315 

316 # Get context IDs 

317 request_id = get_request_id() 

318 session_id = get_session_id() 

319 

320 # For tests where session_id isn't set, use object id as fallback 

321 if not session_id: 

322 session_id = f"test-{id(init_result)}" 

323 

324 # Only log if we haven't logged initialize for this session yet 

325 if session_id not in LoggingProxyClient._logged_sessions_global: 

326 LoggingProxyClient._logged_sessions_global.add(session_id) 

327 

328 # Extract metadata 

329 metadata = extract_initialize_metadata(init_result, self._system_logger) 

330 

331 # Create Pydantic event (use original session_id from context, which may be None) 

332 event = BackendResponseEvent( 

333 transport=self._transport, 

334 method="initialize", 

335 duration_ms=0.0, # Duration not available for auto-initialize 

336 request_id=request_id, 

337 session_id=get_session_id(), # Original from context (may be None for tests) 

338 **metadata, 

339 ) 

340 

341 self._logger.info(event.model_dump(exclude_none=True)) 

342 

343 except Exception as e: 

344 # Log failure to system logger but don't break connection 

345 # This is debug logging, not audit - failure is not security-critical 

346 self._system_logger.warning( 

347 { 

348 "event": "initialize_logging_failed", 

349 "error": str(e), 

350 "error_type": type(e).__name__, 

351 "message": "Failed to log initialize handshake - debug trail incomplete", 

352 } 

353 ) 

354 

355 return self

371 def new(self) -> "LoggingProxyClient": 

372 """Create a new logging client instance for session isolation.

374 This is called by FastMCP.as_proxy() for session isolation.

377 Returns:

397 def __getattr__(self, name: str) -> Any: 

398 """Forward unknown attributes to wrapped client in wrapper mode."""

405def create_logging_proxy_client(

409 debug_enabled: bool = True, 

410) -> LoggingProxyClient: 

411 """Create a LoggingProxyClient for Proxy↔Backend wire-level debugging. 

412 

413 This sets up the backend logger and creates a ProxyClient with logging enabled 

414 for all method calls, arguments, responses, and errors using Pydantic models. 

415 

416 Security-focused logging extracts: 

417 - Tool names and operation types 

418 - File paths and metadata (size, extension, mime type) 

419 - Server information (name, version, capabilities) 

420 - NO file contents or hashes 

421 

422 Args: 

423 transport: Transport object (StdioTransport, HttpTransport, etc.) 

424 log_path: Path to the backend wire log file (from config via get_backend_log_path()). 

425 transport_type: Transport type being used ("stdio" or "streamablehttp").

427

430 """ 

431 if debug_enabled: 

432 logger = setup_backend_wire_logger(log_path) 

433 else: 

434 # Create a logger that discards all messages 

435 logger = logging.getLogger("mcp-acp-core.debug.backend.null") 

436 logger.handlers.clear() 

437 logger.addHandler(logging.NullHandler()) 

438 logger.propagate = False 

439 

440 return LoggingProxyClient(transport, logger, transport_type)
