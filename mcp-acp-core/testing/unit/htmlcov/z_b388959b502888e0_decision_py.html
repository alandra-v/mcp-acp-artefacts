1"""Pydantic models for policy decision logs. 

2 

3IMPORTANT: The 'time' field is Optional[str] = None because: 

4- Model instances are created WITHOUT timestamps (time=None) 

5- ISO8601Formatter adds the timestamp during log serialization 

6- This provides a single source of truth for timestamps 

7 

8Decision events are logged to audit/decisions.jsonl for every policy evaluation, 

9including discovery method bypasses. 

10""" 

11 

12from __future__ import annotations 

13 

14from typing import Any, Literal, Optional 

15 

16from pydantic import BaseModel, ConfigDict, Field 

17 

18 

19class DecisionEvent(BaseModel): 

20 """One policy decision log entry (audit/decisions.jsonl). 

21 

22 Logged on every policy evaluation for forensics and debugging. 

23 Includes discovery bypasses for complete audit trail. 

24 

25 Note: 'time' is None when created, populated by ISO8601Formatter during logging. 

26 """ 

27 

28 # --- core --- 

29 time: Optional[str] = Field( 

30 None, 

31 description="ISO 8601 timestamp, added by formatter during serialization", 

32 ) 

33 event: Literal["policy_decision"] = "policy_decision" 

34 

35 # --- decision outcome --- 

36 decision: Literal["allow", "deny", "hitl"] 

37 matched_rules: list[str] = Field(default_factory=list) 

38 final_rule: str # Rule ID that determined outcome, or "default", "discovery_bypass" 

39 

40 # --- context summary (not full context for privacy) --- 

41 mcp_method: str 

42 tool_name: Optional[str] = None 

43 path: Optional[str] = None # File path (from tool arguments) 

44 uri: Optional[str] = None # Resource URI (from resources/read) 

45 scheme: Optional[str] = None # URI scheme (file, https, s3, etc.) 

46 subject_id: Optional[str] = None # Optional until auth is fully implemented 

47 backend_id: str # Backend server ID (always known from config) 

48 is_mutating: bool = False # Whether the action is mutating 

49 side_effects: Optional[list[str]] = None # Tool side effects (FS_WRITE, CODE_EXEC, etc.) 

50 

51 # --- policy --- 

52 policy_version: str # Policy version for replay/forensics (always loaded) 

53 

54 # --- performance --- 

55 policy_eval_ms: float # Policy rule evaluation time 

56 policy_hitl_ms: Optional[float] = None # HITL wait time (only for HITL decisions) 

57 policy_total_ms: float # Total evaluation time (eval + HITL) 

58 

59 # --- correlation --- 

60 request_id: str # JSON-RPC request ID (every decision has a request) 

61 session_id: Optional[str] = None # May not exist during initialize 

62 

63 # --- HITL-specific fields (only when decision == "hitl") --- 

64 hitl_outcome: Optional[Literal["user_allowed", "user_denied", "timeout"]] = None 

65 

66 model_config = ConfigDict(extra="forbid")
