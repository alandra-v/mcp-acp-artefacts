1"""Proxy server implementation using FastMCP and ProxyClient. 

2 

3This module provides the core proxy functionality that forwards all MCP requests 

4from clients (via STDIO) to backend servers with bidirectional logging. 

5 

6Supported backend transports: 

7- STDIO: Spawns backend server as a child process 

8- Streamable HTTP: Connects to an HTTP endpoint 

9 

10This module is designed for programmatic use: 

11- Import create_proxy(config) to integrate into applications or CLI 

12- The CLI (cli.py) handles user interaction, startup messages, and error display 

13 

14Security: 

15- Audit logs use fail-closed handlers that trigger shutdown if compromised 

16- Startup validation ensures audit logs are writable before accepting requests 

17"""

56def create_proxy( 

57 config: AppConfig, 

58 config_version: str | None = None, 

59 policy_version: str | None = None, 

60) -> tuple[FastMCP, str]: 

61 """Create a transparent proxy that forwards all requests to backend. 

62 

63 This function creates a FastMCP proxy server using ProxyClient to connect 

64 to a backend MCP server using the provided configuration. 

65 

66 Transport selection (handled by create_backend_transport): 

67 - If config.backend.transport is set, use that transport (fail if unavailable) 

68 - If config.backend.transport is None, auto-detect: 

69 - Prefer Streamable HTTP if configured and reachable 

70 - Fall back to STDIO if HTTP unavailable or not configured 

71 

72 Logging: 

73 - Audit logs (ALWAYS enabled): <log_dir>/mcp_acp_core_logs/audit/operations.jsonl 

74 - Decision logs (ALWAYS enabled): <log_dir>/mcp_acp_core_logs/audit/decisions.jsonl 

75 - Debug wire logs (when log_level == "DEBUG"): 

76 - Client<->Proxy: <log_dir>/mcp_acp_core_logs/debug/client_wire.jsonl 

77 - Proxy<->Backend: <log_dir>/mcp_acp_core_logs/debug/backend_wire.jsonl 

78 - All logs include correlation IDs (request_id, session_id) 

79 - JSONL format with ISO 8601 timestamps 

80 

81 Middleware order (outer to inner): 

82 - Context: Sets up request context (request_id, session_id, tool_context) 

83 - Audit: Logs all operations including denials 

84 - Client logging: Wire-level debugging 

85 - Enforcement: Policy evaluation and blocking (innermost) 

86 

87 Args: 

88 config: Application configuration loaded from mcp_acp_core_config.json. 

89 config_version: Current config version from config history (e.g., "v1"). 

90 policy_version: Current policy version from policy history (e.g., "v1"). 

91 

92 Returns: 

93 Tuple of (FastMCP proxy instance, actual transport type used). 

94 Transport type is "stdio" or "streamablehttp". 

95 

96 Raises: 

97 ValueError: If transport config is missing for selected transport. 

98 FileNotFoundError: If STDIO backend command is not found in PATH. 

99 PermissionError: If insufficient permissions to execute backend command. 

100 TimeoutError: If HTTP backend connection times out. 

101 ConnectionError: If HTTP backend is unreachable. 

102 RuntimeError: If backend server fails to start or initialize. 

103 AuditFailure: If audit logs cannot be written at startup. 

104 """ 

105 # Configure system logger file handler with user's log_dir 

106 configure_system_logger_file(get_system_log_path(config)) 

107 

108 # Validate audit logs are writable BEFORE starting 

109 # If this fails, we raise AuditFailure and don't start 

110 audit_path = get_audit_log_path(config) 

111 decisions_path = get_decisions_log_path(config) 

112 verify_audit_writable(audit_path) 

113 verify_audit_writable(decisions_path) 

114 

115 # Create shutdown coordinator for fail-closed behavior 

116 log_dir = get_log_dir(config) 

117 system_logger = get_system_logger() 

118 shutdown_coordinator = ShutdownCoordinator(log_dir, system_logger) 

119 

120 # Create shutdown callback with hybrid approach: 

121 # Try async coordinator if event loop is running, fall back to sync 

122 def on_audit_failure(reason: str) -> None: 

123 """Handle audit log integrity failure.""" 

124 try: 

125 loop = asyncio.get_running_loop() 

126 loop.create_task( 

127 shutdown_coordinator.initiate_shutdown( 

128 failure_type=AuditFailure.failure_type, 

129 reason=reason, 

130 exit_code=AuditFailure.exit_code, 

131 context={"source": "audit_handler"}, 

132 ) 

133 ) 

134 except RuntimeError: 

135 # No event loop running - use sync fallback 

136 sync_emergency_shutdown( 

137 log_dir, AuditFailure.failure_type, reason, exit_code=AuditFailure.exit_code 

138 ) 

139 

140 # Create backend transport (handles detection, validation, health checks) 

141 transport, transport_type = create_backend_transport(config.backend) 

142 

143 # Determine if debug wire logging is enabled 

144 debug_enabled = config.logging.log_level == "DEBUG" 

145 

146 # Create LoggingProxyClient with transport (logs to backend_wire.jsonl) 

147 logging_backend_client = create_logging_proxy_client( 

148 transport, 

149 log_path=get_backend_log_path(config), 

150 transport_type=transport_type, 

151 debug_enabled=debug_enabled, 

152 ) 

153 

154 # Create proxy with logging-wrapped backend client 

155 proxy = FastMCP.as_proxy( 

156 logging_backend_client, 

157 name=config.proxy.name, 

158 ) 

159 

160 # Create identity provider (Stage 1: local user via getpass.getuser()) 

161 identity_provider = create_identity_provider() 

162 

163 # Register context middleware (outermost - added first) 

164 # Sets up request_id, session_id, and tool_context for all downstream middleware 

165 context_middleware = create_context_middleware() 

166 proxy.add_middleware(context_middleware) 

167 

168 # Register audit logging middleware (ALWAYS enabled) 

169 # Logs single event per operation to audit/operations.jsonl 

170 # Uses fail-closed handler that triggers shutdown if log is compromised 

171 audit_middleware = create_audit_logging_middleware( 

172 log_path=audit_path, 

173 shutdown_coordinator=shutdown_coordinator, 

174 shutdown_callback=on_audit_failure, 

175 backend_id=config.backend.server_name, 

176 identity_provider=identity_provider, 

177 transport=transport_type, 

178 config_version=config_version, 

179 ) 

180 proxy.add_middleware(audit_middleware) 

181 

182 # Register client logging middleware (logs to client_wire.jsonl) 

183 # 

184 # Note: We intentionally don't use FastMCP's ErrorHandlingMiddleware here. 

185 # Backend MCP errors are already properly formatted and forwarded as-is. 

186 # Proxy-level errors (transport failures, internal errors) surface as raw 

187 # exceptions - this is acceptable since they're rare and the raw messages 

188 # provide useful diagnostic context. See docs/architecture.md for details. 

189 client_middleware = create_client_logging_middleware( 

190 log_path=get_client_log_path(config), 

191 transport=transport_type, 

192 debug_enabled=debug_enabled, 

193 ) 

194 proxy.add_middleware(client_middleware) 

195 

196 # Load policy and register enforcement middleware (innermost - added last) 

197 # Evaluates policy and blocks denied requests before they reach the backend. 

198 # Logs every decision to audit/decisions.jsonl with fail-closed handler 

199 policy = load_policy() 

200 

201 # Build protected directories tuple (config dir + log dir) 

202 # These paths are protected from MCP tool access - built-in security 

203 # Use os.path.realpath() to resolve ALL symlinks for security 

204 protected_dirs = ( 

205 PROTECTED_CONFIG_DIR, 

206 os.path.realpath(log_dir), 

207 ) 

208 

209 enforcement_middleware = create_enforcement_middleware( 

210 policy=policy, 

211 protected_dirs=protected_dirs, 

212 identity_provider=identity_provider, 

213 backend_id=config.backend.server_name, 

214 log_path=decisions_path, 

215 shutdown_callback=on_audit_failure, 

216 policy_version=policy_version, 

217 ) 

218 proxy.add_middleware(enforcement_middleware) 

219 

220 return proxy, transport_type
